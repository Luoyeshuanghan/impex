开始

impex.js 的设计目标是让每个对构建web应用感兴趣的人都可以使用，所以从接口设计到语法支持 impex.js 都保持简单的写法

Hello Impex 


<div id="entry">
  Hello {{ name }} !
</div>

// 组件 
var debug = impex.render(document.getElementById('entry'),{
	name:'impex'
});
Hello {{name}} !

双向绑定 


<div id="entry">
  Hello {{ name }} !
  <input x-model="name" value="{{name}}">
</div>

// 组件 
var debug = impex.render(document.getElementById('entry'),{
	name:'impex'
});
Hello {{name}} !

交互 


.bigger p{
	font-size:2rem;
}


<div id="entry" class="{{bigger?'bigger':''}}">
  <p> Hello impex ! </p>
  <button x-click="toggleSize()"> toggle size </button>
</div>

// 组件 
var debug = impex.render(document.getElementById('entry'),{
	bigger:false,
	toggleSize:function(){
		this.bigger = !this.bigger;
	}
});

Hello impex !
